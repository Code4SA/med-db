select country.name, count(DISTINCT medicine.medicine_id) from procurement
join country using (country_id)
join product using (product_id)
join medicine using (medicine_id)
group by procurement.country_id 
order by country.name;

select country.name, count(DISTINCT procurement.procurement_id) from procurement
join country using (country_id)
join product using (product_id)
join medicine using (medicine_id)
where product.manufacturer_id is not Null and product.manufacturer_id != 4
group by procurement.country_id
order by country.name;
